YOLOv8 + DeepSORT (Python)
"Mataloob" ~ (YOLOv8 detects people in each frame, and DeepSORT tracks them by assigning unique IDs, enabling accurate crowd counting without duplication)

pip install ultralytics deep-sort-realtime opencv-python

#2(Import Required Modules)

import cv2
from ultralytics import YOLO
from deep_sort_realtime.deepsort_tracker import DeepSort

#3(Load YOLOv8 Model)

model = YOLO("yolov8n.pt")

#4(Initialize DeepSORT Tracker)

tracker = DeepSort(max_age=30)

#5(Read Video)

cap = cv2.VideoCapture("crowd.mp4")
unique_ids = set()

#6(Detect People (YOLOv8))

while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        break

    results = model(frame)[0]
    detections = [ ]

#7(Convert YOLO Output for DeepSORT)

    for box in results.boxes:
        cls_id = int(box.cls[0])
        conf = float(box.conf[0])

        if cls_id == 0 and conf > 0.5:   # person
            x1, y1, x2, y2 = map(int, box.xyxy[0])
            detections.append(([x1, y1, x2-x1, y2-y1], conf, "person"))

#8(Apply DeepSORT Tracking)

    tracks = tracker.update_tracks(detections, frame=frame)

#9(Draw Boxes, IDs & Count People)


    for track in tracks:
        if not track.is_confirmed():
            continue

        track_id = track.track_id
        unique_ids.add(track_id)

        l, t, r, b = map(int, track.to_ltrb())
        cv2.rectangle(frame, (l, t), (r, b), (0,255,0), 2)
        cv2.putText(frame, f"ID {track_id}", (l, t-10),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0,255,0), 2)

#10(Display Total Count)

    cv2.putText(frame, f"Total People: {len(unique_ids)}",
                (20,40), cv2.FONT_HERSHEY_SIMPLEX, 1,
                (0,0,255), 2)

#LAST BUT NOT LEAST (show output)


    cv2.imshow("YOLOv8 + DeepSORT", frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break


//and release resources 

cap.release()
cv2.destroyAllWindows()
